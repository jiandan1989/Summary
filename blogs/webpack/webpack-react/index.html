Webpack + React

初始化

# 初始化
npm init -y

# 安装 webpack
npm install -D webpack webpack-cli webpack-dev-server

# 安装 babel
npm install -D babel-loader @babel/core @babel/preset-env @babel/preset-react

# 创建文件
mkdir src && cd src touch index.js
touch webpack.config.js

babel

在根目录下创建 .babelrc, babel: preset-env 解析 ES6 为 ES5, preset-react: 解析 jsx 文件

{
  "presets": ["@babel/preset-env", "@babel/preset-react"]
}

添加 webpack.config.js

// webpack.config.js
module.exports = {
  module: {
    rules: [
      {
        test: /\.(js|jsx)$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
        },
      },
    ],
  },
};

React

npm install -D react react-dom

HTML Plugin

npm install html-webpack-plugin html-loader --save-dev

添加 HTML 模板

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta http-equiv="X-UA-Compatible" content="ie=edge" />
    <title>Document</title>
  </head>
  <body>
    <div id="root"></div>
  </body>
</html>

再次修改 webpack.config.js

const HtmlWebpackPlugin = require('html-webpack-plugin');
module.exports = {
  module: {
    rules: [
      {
        test: /\.(js|jsx)$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
        },
      },
      {
        test: /\.html$/,
        use: {
          loader: 'html-loader',
          options: { minimize: true },
        },
      },
    ],
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: './template/index.html',
      filename: './index.html',
    }),
  ],
};

package.json

在 webpackv4 中 webpack 自带两种模式 development: 开发模式, production: 生产模式, 设置 mode 之后将会设置 process.env.NODE_ENV 的 DefinePlugin

"script": {
  "dev": "webpack --mode development",
  "build": "webpack --mode production",
}

启动服务

npm i --save-dev webpack-dev-server

修改 package.json

"scripts": {
  "start": "webpack-dev-server --mode development --open"
}

其他配置

const path = require('path');
module.exports = {

  devServer: {

    after(app, server) {
      // 启动之后

当以上配置完毕后, 可以开始编写 React 代码了

Component

src/index.js

import React from 'react';
import { render } from 'react-dom';
function App() {
  return <div>React</div>;
}
render(<App />, document.getElementById('root'));

启动 打开 localhost:9000

npm run dev

试着修改 index.js

function App() {
  return <div>Hello React!</div>;
}

到此之后, 无需刷新页面即可可以看到最新的页面中显示 Hello React

打包 build

npm run build

可以在目录下新生成一个 dist 文件夹, 打开之后会有新生成的 index.html 以及打包后的 main.js

在以前, 提取文件是 extract-text-webpack-plugin 的事, 但现在与 webpack V4 不是很匹配,所以此处使用 mini-css-extract-plugin 替代

npm i css-loader min-css-extract-plugin --save-dev

添加 css 文件

// src/index.css
body {
  margin: 0;
  background: red;
  color: #fff;
  font-weight: 900;
}
// index.js
import './index.css';
// webpack.config.js
const MiniCssExtractPlugin = require('mini-css-extract-plugin');
module.exports = {
  module: {
    rules: [
      // ...
      {
        test: /\.css$/,
        use: [
          {
            loader: MiniCssExtractPlugin.loader,
            hmr: process.env.NODE_ENV === 'development',
          },
          'css-loader',
        ],
      },
    ],
  },
  plugins: [
    // ...
    new MiniCssExtractPlugin({
      filename: '[name].css',
      chunkFilename: '[id].css',
    }),
  ],
};
